function solution(nums) {
	let n=nums.length;
	for(let i=0; i<n-1; i++){
		let minIndex=i;
		for(let j=i+1; j<n; j++){
			if(nums[j]<nums[minIndex]) minIndex=j;
		}
		if(i!==minIndex){
			[nums[i], nums[minIndex]] = [nums[minIndex], nums[i]];	
		}
	}
	return nums;        
}    
console.log(solution([2, 3, 1, 5, 4]));
